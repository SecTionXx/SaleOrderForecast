Modal Component

Basic Modal

// Create a modal
const modal = new Modal({
  title: 'Example Modal',
  content: 'This is a simple modal dialog.',
  buttons: [
    {
      text: 'Close',
      variant: 'secondary',
      onClick: () => console.log('Modal closed')
    },
    {
      text: 'Save Changes',
      variant: 'primary',
      onClick: () => console.log('Changes saved')
    }
  ]
});

// Show the modal
document.getElementById('open-modal-btn').addEventListener('click', () => {
  modal.show();
});

Table Component

Data Table

// Sample data
const data = [
  { id: 1, name: 'John Doe', email: 'john@example.com', status: 'Active' },
  { id: 2, name: 'Jane Smith', email: 'jane@example.com', status: 'Inactive' },
  { id: 3, name: 'Bob Johnson', email: 'bob@example.com', status: 'Active' },
  { id: 4, name: 'Alice Brown', email: 'alice@example.com', status: 'Pending' }
];

// Create table
const table = new Table({
  columns: [
    { key: 'id', title: 'ID', sortable: true },
    { key: 'name', title: 'Name', sortable: true },
    { key: 'email', title: 'Email' },
    { 
      key: 'status', 
      title: 'Status',
      render: (value) => {
        const statusClass = value.toLowerCase();
        return `${value}`;
      }
    }
  ],
  data: data,
  paginate: true,
  pageSize: 2
});

document.getElementById('table-container').appendChild(table.element);
